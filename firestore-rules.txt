rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Convite: resolve código -> casa. Só GET direto (quem tem o código);
    // ninguém consegue LISTAR os códigos existentes.
    match /convites/{codigo} {
      allow get: if request.auth != null;
      allow list: if false;
      allow create: if request.auth != null;
    }

    match /casas/{casaId} {
      allow create: if request.auth != null;
      // Só membros leem a casa; sem listagem (impede enumerar códigos)
      allow get: if request.auth != null
        && request.auth.uid in resource.data.membros;
      allow list: if false;
      allow update: if request.auth != null
        && (request.auth.uid in resource.data.membros
            || request.auth.uid in request.resource.data.membros);

      function ehMembro() {
        return request.auth != null
          && request.auth.uid in
             get(/databases/$(database)/documents/casas/$(casaId)).data.membros;
      }

      // Lançamentos da Casa: qualquer membro lê; só o AUTOR edita/apaga.
      // Docs antigos sem criadoPorUid continuam editáveis por todos.
      match /transacoes/{transacaoId} {
        allow read: if ehMembro();
        allow create: if ehMembro()
          && (!("criadoPorUid" in request.resource.data)
              || request.resource.data.criadoPorUid == ""
              || request.resource.data.criadoPorUid == request.auth.uid);
        allow update, delete: if ehMembro()
          && (!("criadoPorUid" in resource.data)
              || resource.data.criadoPorUid == ""
              || resource.data.criadoPorUid == request.auth.uid);
      }

      // Categorias são coletivas: qualquer membro gerencia.
      match /categorias/{categoriaId} {
        allow read, write: if ehMembro();
      }

      // Espelho da visão Membros: cada membro só escreve no próprio feed.
      match /membros/{membroUid}/{documento=**} {
        allow read: if ehMembro();
        allow write: if ehMembro() && request.auth.uid == membroUid;
      }
    }

    // Dados privados do usuário: sync entre aparelhos e backup na nuvem
    match /usuarios/{uid}/{documento=**} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }
  }
}

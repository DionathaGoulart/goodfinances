rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Convite: resolve código -> casa. Só GET direto (quem tem o código);
    // ninguém consegue LISTAR os códigos existentes. Criar um convite exige
    // já ser membro da casa apontada — senão qualquer autenticado poderia
    // registrar códigos apontando para a casa dos outros (squatting/phishing).
    match /convites/{codigo} {
      allow get: if request.auth != null;
      allow list: if false;
      allow create: if request.auth != null
        && exists(/databases/$(database)/documents/casas/$(request.resource.data.casaId))
        && request.auth.uid in
           get(/databases/$(database)/documents/casas/$(request.resource.data.casaId)).data.membros;
      allow update, delete: if false;
    }

    match /casas/{casaId} {
      // Ao criar, o dono precisa se incluir nos membros e ser o criadoPor.
      allow create: if request.auth != null
        && request.auth.uid in request.resource.data.membros
        && request.resource.data.criadoPor == request.auth.uid;
      // Só membros leem a casa; sem listagem (impede enumerar códigos)
      allow get: if request.auth != null
        && request.auth.uid in resource.data.membros;
      allow list: if false;
      allow update: if request.auth != null
        && (
          // Membro existente edita (sair da casa, auto-reparo do convite).
          request.auth.uid in resource.data.membros
          // Ou entrando pelo convite: NÃO era membro, o novo estado só
          // ACRESCENTA o próprio uid (mantém todos os atuais) e não mexe em
          // código de convite nem no criador. Sem isso, qualquer autenticado
          // que descobrisse o casaId trocaria a lista de membros inteira.
          || (
            !(request.auth.uid in resource.data.membros)
            && request.auth.uid in request.resource.data.membros
            && request.resource.data.membros.hasAll(resource.data.membros)
            && request.resource.data.membros.size() == resource.data.membros.size() + 1
            && request.resource.data.codigoConvite == resource.data.codigoConvite
            && request.resource.data.criadoPor == resource.data.criadoPor
          )
        );

      function ehMembro() {
        return request.auth != null
          && request.auth.uid in
             get(/databases/$(database)/documents/casas/$(casaId)).data.membros;
      }

      // Lançamentos da Casa: qualquer membro lê; só o AUTOR edita/apaga.
      // Docs antigos sem criadoPorUid continuam editáveis por todos.
      match /transacoes/{transacaoId} {
        allow read: if ehMembro();
        allow create: if ehMembro()
          && (!("criadoPorUid" in request.resource.data)
              || request.resource.data.criadoPorUid == ""
              || request.resource.data.criadoPorUid == request.auth.uid);
        allow update, delete: if ehMembro()
          && (!("criadoPorUid" in resource.data)
              || resource.data.criadoPorUid == ""
              || resource.data.criadoPorUid == request.auth.uid);
      }

      // Categorias são coletivas: qualquer membro gerencia.
      match /categorias/{categoriaId} {
        allow read, write: if ehMembro();
      }

      // Cartões, metas e contas da Casa: coletivos (qualquer membro gerencia).
      match /cartoes/{cartaoId} {
        allow read, write: if ehMembro();
      }
      match /metas/{metaId} {
        allow read, write: if ehMembro();
      }
      match /contas/{contaId} {
        allow read, write: if ehMembro();
      }

      // Espelho da visão Membros: cada membro só escreve no próprio feed.
      match /membros/{membroUid}/{documento=**} {
        allow read: if ehMembro();
        allow write: if ehMembro() && request.auth.uid == membroUid;
      }
    }

    // Dados privados do usuário: sync entre aparelhos e backup na nuvem
    match /usuarios/{uid}/{documento=**} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }
  }
}
